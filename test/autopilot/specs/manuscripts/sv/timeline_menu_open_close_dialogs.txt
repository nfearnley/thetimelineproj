# Timeline menu

# Open a well known Timeline
Select Menu(Hjälp|Help, "Getting started tutorial") 

Select Menu(Tidslinje|Timeline, "Skapa Händelse...|Create Event...") 
Click Button(Avbryt)

# Make sure an event is selected with this command
click mouse(500,570)

Select Menu(Tidslinje|Timeline, "Editera valda Händelser...|Edit Selected Event...") 
Click Button(Avbryt)

Select Menu(Tidslinje|Timeline, "Duplicera Markerad Händelse...|Duplicate Selected Event...") 
Click Button(Avbryt)

Select Menu(Tidslinje|Timeline, "Sätt Kategori på valda Händelser...|Set Category on Selected Events...") 
Click Button(Avbryt)

# Click deselects other selected events!
click mouse(1400,570,Ctrl)
Select Menu(Tidslinje|Timeline, "Mät avståndet mellan två valda händelser...|Measure Distance between two Events...") 
Click Button(OK)

Select Menu(Tidslinje|Timeline, "Set Category on events without category...") 
Click Button(Avbryt)

Select Menu(Tidslinje|Timeline, "Redigera Kategorier|Edit Categories") 
Click Button(Stäng)

Select Menu(Tidslinje|Timeline, "Read Only")
